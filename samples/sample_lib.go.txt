package main 

import (
	"fmt"

	dalfox "github.com/hahwul/dalfox/v2/lib"
)

func main() {
	opt := dalfox.Options{
		Cookie:     "ABCD=1234",
		NoBAV:      true,
		Mining:     false,
		FindingDOM: false,
		UniqParam:  "abababab",
	}
	result, _ := dalfox.NewScan(dalfox.Target{
		URL:     "https://www.hahwul.com?abababab=1",
		Method:  "GET",
		Options: opt,
	})
	fmt.Println(result)
}
